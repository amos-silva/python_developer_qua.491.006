instalar biblioteca flet
pip install flet

criar projeto
flet create "nome do projeto"

executar o programa
flet run "nome do projeto"

executar programa na web
flet run --web "nome do projeto"

# criar programa executavel com icone
flet pack 05_imc/src/main.py --name=app_imc --icon 05_imc/src/assets/1.png
